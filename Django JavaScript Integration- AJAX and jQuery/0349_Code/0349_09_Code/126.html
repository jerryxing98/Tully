PHOTO_DIRECTORY.load_profile = function(id) { PHOTO_DIRECTORY.current_profile = id; try { $("#profile").html(PHOTO_DIRECTORY.build_profile(id)); PHOTO_DIRECTORY.register_update(); } catch(error) { if (PHOTO_DIRECTORY.current_profile) { $("#profile").load("/ajax/profile/" + PHOTO_DIRECTORY.current_profile, PHOTO_DIRECTORY.register_update); } else { $("#profile").html(""); } } }
